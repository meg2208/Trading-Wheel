/* Highest Value any portfolio reaches */ 
SELECT CS.user_id, S.strategy_name, T.portfolio_value 

	FROM trade T, action A, create_strategy CS, strategy S
	
	WHERE CS.strategy_id = A.strategy_id AND S.strategy_id = CS.strategy_id AND T.trade_id = A.trade_id

	GROUP BY CS.user_id, S.strategy_name, T.portfolio_value
	
	HAVING MAX(T.portfolio_value)



/* Number of trades made by each strategy for each person */

Select CS.strategy_name, CS.user_id, COUNT(T.trade_id)
	
	FROM create_strategy CS, trade T

	WHERE CS.strategy_id = T.strategy_id AND CS.trade_id = T.trade_id

	GROUP BY CS.strategy_id, CS.user_id


/* Volume of each stock in 2009, in order */

Select Distinct Q.security, SUM(Q.volume) as '09 Volume
	
	FROM query_data Q

	WHERE YEAR(Q.time) = 2009

	GROUP BY Q.security
